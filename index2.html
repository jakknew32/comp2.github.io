📡 ตั้งค่า Geocache (Google Sheets)

บันทึกพิกัดลง Google Sheets ให้ทุกคนใช้ cache ร่วมกัน — ไม่ต้อง geocode ซ้ำ ลด API call

Sheet ID (ไฟล์ข้อมูลหลัก)
ใช้ Sheet เดียวกับข้อมูล Order — ระบบจะสร้าง Sheet ชื่อ GeoCache ให้อัตโนมัติ
Apps Script Web App URL (สำหรับเขียนข้อมูล)
ต้องสร้าง Apps Script ก่อน — คัดลอก code ด้านล่างแล้ว Deploy เป็น Web App
📋 วิธีสร้าง Apps Script (คลิกเพื่อดู)
1. เปิด Google Sheets → Extensions → Apps Script → วาง code นี้ → Deploy → New deployment → Web App → Execute as: Me → Anyone can access
function doPost(e){ try{ const d=JSON.parse(e.postData.contents); if(d.action==='appendGeoCache'){ const ss=SpreadsheetApp.getActiveSpreadsheet(); let sh=ss.getSheetByName('GeoCache'); if(!sh){ sh=ss.insertSheet('GeoCache'); sh.appendRow(['address_key','lat','lon','updated']); } // อ่าน keys ที่มีอยู่แล้ว const existing=new Set(); const vals=sh.getDataRange().getValues(); vals.forEach(r=>existing.add(String(r[0]))); // เพิ่มเฉพาะ rows ใหม่ d.rows.forEach(r=>{ if(!existing.has(String(r[0]))){ sh.appendRow([r[0],r[1],r[2],new Date().toISOString()]); existing.add(String(r[0])); } }); return ContentService.createTextOutput(JSON.stringify({ok:true,added:d.rows.length})).setMimeType(ContentService.MimeType.JSON); } }catch(err){ return ContentService.createTextOutput(JSON.stringify({ok:false,error:err.message})).setMimeType(ContentService.MimeType.JSON); } }
